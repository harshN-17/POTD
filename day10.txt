class Solution {
public:
    int countGoodSubstrings(string s) {
        int n=s.size(),dup=0, cnt=0;
        if(n<3) return cnt;
        unordered_map<char,int> mp;
        for(int i=0;i<n;++i){
            mp[s[i]]++;
            if(mp[s[i]]==2) dup++;
            
            if(i>1){
                if(dup==0) cnt++;
                
                if(--mp[s[i-2]]==1) dup--;
            }
        }
        return cnt;
    }
};
