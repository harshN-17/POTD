class Solution {
public:
    bool isPalindrome(string s) {
        string ns="";
        int l = s.size();
        for(int i=0; i<l; ++i){
            if((tolower(s[i])>='a' && tolower(s[i])<='z') or (s[i]>='0' and s[i]<='9'))
               ns += tolower(s[i]);
        }
        int i=0;
        l = ns.size()-1;
        while(i<l){
            if(ns[i++] != ns[l--])
                return false;
        }
        return true;
    }
};
