class Solution {
public:
    int mySqrt(int x) {
        int st=0, en=x;
        long long mid,ans=0;
        while(st<=en){
            mid = st + (en-st)/2;
            if(mid*mid==x) return mid;
            else if(mid*mid>x)en=mid-1;
            else{
                ans=mid;
                st = mid+1;
            }
        }
        return ans;
    }
};
