class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
        int n1 = nums1.size(), n2= nums2.size();
        int right=n1-1;
        int i=n1-n2-1, j=n2-1;
        while(i>=0&&j>=0){
            nums1[right--]= (nums1[i]>=nums2[j]) ? nums1[i--] : nums2[j--];
        }
        for(;i>=0;--i)
            nums1[right--]=nums1[i];
        for(;j>=0;--j)
            nums1[right--]=nums2[j];
    }
    
};
